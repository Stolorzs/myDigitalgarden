```java
public class MulForExercise01{
	public static void main(String[] args){
		Scanner myScanner = new Scanner(System.in);
		double sum = 0;
		double singleSum = 0;//单个班的平均分
		for(int i = 1; i <= 3; i++){
			for(int j = 1; j <= 5; j++){
				System.out.println("请输入第" + i + "班的第" + j 
				+"位同学的成绩(0-100)"); 
				double score = myScanner.nextDouble();
				sum += score;
				singleSum += score;
				score = 0;
			}
			System.out.println("第" + i + "班的平均分为" + (singleSum/5));
			singleSum = 0;
		}
		System.out.println("所有班级的平均分为" + (sum / 15));
	}
}
```

```java
//空心金字塔 
//1    *     4个空格1星 + 0个空格 + 1个星 
//2   * *    3个空格1星 + 1个空格
//3  *   *   2个空格1星 + 3个空格
//4 *     *  1个空格1星 + 5个空格
//5********* 0个空格1星 + 7个星星
//n层
import java.util.Scanner;
public class Stars{
	public static void main(String[] args){
		Scanner myScanner = new Scanner(System.in);
		System.out.println("请输入层数");
		int n = myScanner.nextInt();
		for(int i = 1; i <= n; i++){
			//输出*之前，输出对应空格
			for(int k = 1; k <= n - i; k++){
				System.out.print(" ");
			}
			//固定的星星
			System.out.print("*");
			//控制打印每层内部空格的个数
			if(i < n){
				for(int j = 1; j <= 2 * (i - 1) - 1; j++){
					System.out.print(" ");
				}
			}else if(i == n){
				for(int j = 1; j <= 2 * (i - 1) - 1; j++){
					System.out.print("*");
				}
			}
			//最后一个星星，第一层不打
			if(i != 1){
				System.out.print("*");
			}
			//每打印完一层的*形后换行
			System.out.print("\n");
		}
	}
}
```
